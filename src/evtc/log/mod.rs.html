12/// Log started, ended or target changed.
13#[derive(Debug, Clone)]
14#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
15pub struct LogEvent {
16    /// Time of registering the event.
17    pub time: u64,
18
19    /// Server Unix timestamp.
20    pub server_time: u32,
21
22    /// Local Unix timestamp.
23    pub local_time: u32,
24
25    /// Target species id (or ArcDPS id `0x637261`).
26    pub id: u64,
27}
28
29impl Extract for LogEvent {
30    #[inline]
31    unsafe fn extract(event: &Event) -> Self {
32        Self {
33            time: event.time,
34            server_time: transmute::<i32, u32>(event.value),
35            local_time: transmute::<i32, u32>(event.buff_dmg),
36            id: event.src_agent,
37        }
38    }
39}
40
41impl TryExtract for LogEvent {
42    #[inline]
43    fn can_extract(event: &Event) -> bool {
44        matches!(
45            event.get_statechange(),
46            StateChange::SquadCombatStart | StateChange::SquadCombatEnd | StateChange::LogNPCUpdate
47        )
48    }
49}
