14/// Activation (skill cast) event.
15///
16/// Only animated skill casts are captured by ArcDPS.
17#[derive(Debug, Clone)]
18#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
19pub struct ActivationEvent {
20    /// Time of registering the activation.
21    pub time: u64,
22
23    /// Agent casting the skill.
24    pub agent: AgentId,
25
26    /// Id of casted skill.
27    pub skill_id: u32,
28
29    /// Kind of activation state change.
30    pub activation: Activation,
31
32    /// Activation duration.
33    pub duration: i32,
34
35    /// Scaled activation duration.
36    pub scaled_duration: i32,
37
38    /// Target location, if applicable.
39    pub target: Position,
40}
41
42impl Extract for ActivationEvent {
43    #[inline]
44    unsafe fn extract(event: &Event) -> Self {
45        let [x, y] = transmute_field!(event.dst_agent as [f32; 2]);
46        let z = f32::from_bits(event.overstack_value);
47        Self {
48            time: event.time,
49            agent: AgentId::from_src(event),
50            skill_id: event.skill_id,
51            activation: Activation::from_event(event),
52            duration: event.value,
53            scaled_duration: event.buff_dmg,
54            target: Position::new(x, y, z),
55        }
56    }
57}
58
59impl TryExtract for ActivationEvent {
60    #[inline]
61    fn can_extract(event: &Event) -> bool {
62        LegacyEventCategory::from_event(event) == Some(LegacyEventCategory::Activation)
63    }
64}

66/// Skill activation (cast).
67///
68/// *Arc calls this "combat activation".*
69#[derive(
70    Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, IntoPrimitive, FromPrimitive,
71)]
72#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
73#[cfg_attr(
74    feature = "strum",
75    derive(Display, EnumCount, EnumIter, IntoStaticStr, VariantNames)
76)]
77#[repr(u8)]
78pub enum Activation {
79    /// Not used, different kind of event.
80    None = 0,
81
82    /// Started skill/animation activation.
83    Start = 1,
84
85    /// Unused as of 5th November 2019.
86    QuicknessUnused = 2,
87
88    /// Stopped skill activation with reaching tooltip time.
89    CancelFire = 3,
90
91    /// Stopped skill activation without reaching tooltip time.
92    CancelCancel = 4,
93
94    /// Animation completed fully.
95    Reset = 5,
96
97    /// Unknown or invalid.
98    #[num_enum(catch_all)]
99    Unknown(u8),
100}
101
102impl Activation {
103    #[inline]
104    pub fn from_event(event: &Event) -> Self {
105        event.is_activation.into()
106    }
107}
