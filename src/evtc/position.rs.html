42/// Positional information.
43///
44/// This can be from an [`Event`] with [`StateChange::Position`], [`StateChange::Velocity`] or [`StateChange::Facing`].
45/// It can also occur in effect or missile events as locations or orientations.
46///
47/// Ingame coordinates are interpreted as 1 unit = 1 inch.
48/// The z-axis represents vertical height and **points down**,
49/// meaning lower values are a higher location ingame.
50///
51/// Mumble coordinates are given in meters.
52/// The y-axis represents vertical height and **points up**.
53#[derive(Debug, Clone, PartialEq)]
54#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
55pub struct Position {
56    pub x: f32,
57    pub y: f32,
58    pub z: f32,
59}
60
61impl Position {
62    /// Conversion from inch to meter.
63    pub const INCH_TO_METER: f32 = 0.0254;
64
65    /// Creates new positional information.
66    #[inline]
67    pub const fn new(x: f32, y: f32, z: f32) -> Self {
68        Self { x, y, z }
69    }
70
71    /// Creates a position from Mumble coordinates.
72    #[inline]
73    pub fn from_mumble(coords: [f32; 3]) -> Self {
74        let [x, y, z] = coords;
75        Self::new(
76            x / Self::INCH_TO_METER,
77            z / Self::INCH_TO_METER,
78            -y / Self::INCH_TO_METER,
79        )
80    }
81
82    /// Creates a position from [`i16`] values with a scaling factor.
83    #[inline]
84    pub fn from_scaled_i16s(x: i16, y: i16, z: i16, factor: f32) -> Self {
85        let x = x as f32 * factor;
86        let y = y as f32 * factor;
87        let z = z as f32 * factor;
88        Self::new(x, y, z)
89    }
90
91    /// Converts the position to an [`array`].
92    #[inline]
93    pub fn to_array(self) -> [f32; 3] {
94        [self.x, self.y, self.z]
95    }
96
97    /// Converts the position to a [`tuple`].
98    #[inline]
99    pub fn to_tuple(self) -> (f32, f32, f32) {
100        (self.x, self.y, self.z)
101    }
102
103    /// Converts the position to Mumble coordinates.
104    #[inline]
105    pub fn to_mumble(&self) -> [f32; 3] {
106        [
107            self.x * Self::INCH_TO_METER,
108            -self.z * Self::INCH_TO_METER,
109            self.y * Self::INCH_TO_METER,
110        ]
111    }
112
113    /// Returns the length of the position interpreted as vector.
114    #[inline]
115    pub fn len(&self) -> f32 {
116        (self.x.powi(2) + self.y.powi(2) + self.z.powi(2)).sqrt()
117    }
118
119    /// Interprets the position as vector and multiplies it with the given matrix.
120    #[inline]
121    pub fn mat_mul(&self, matrix: [[f32; 3]; 3]) -> Self {
122        let x = matrix[0][0] * self.x + matrix[0][1] * self.y + matrix[0][2] * self.z;
123        let y = matrix[1][0] * self.x + matrix[1][1] * self.y + matrix[1][2] * self.z;
124        let z = matrix[2][0] * self.x + matrix[2][1] * self.y + matrix[2][2] * self.z;
125        Self::new(x, y, z)
126    }
127
128    /// Interprets the position as rotation angles and converts it to a rotation matrix.
129    ///
130    /// `x`, `y` and `z` are interpreted as angles around each axis in radians.
131    #[inline]
132    pub fn as_rotation_matrix(&self) -> [[f32; 3]; 3] {
133        let Self {
134            x: alpha,
135            y: beta,
136            z: gamma,
137        } = self;
138        [
139            [
140                beta.cos() * gamma.cos(),
141                alpha.sin() * beta.sin() * gamma.cos() - alpha.cos() + gamma.sin(),
142                alpha.cos() * beta.sin() * gamma.cos() + alpha.sin() * gamma.sin(),
143            ],
144            [
145                beta.cos() * gamma.sin(),
146                alpha.sin() * beta.sin() * gamma.sin() + alpha.cos() + gamma.cos(),
147                alpha.cos() * beta.sin() * gamma.sin() - alpha.sin() * gamma.cos(),
148            ],
149            [
150                -beta.sin(),
151                alpha.sin() * beta.cos(),
152                alpha.cos() * beta.cos(),
153            ],
154        ]
155    }
156
157    /// Interprets the position as rotation angles and rotates the given vector.
158    #[inline]
159    pub fn rotate(&self, vector: Self) -> Self {
160        vector.mat_mul(self.as_rotation_matrix())
161    }
162
163    /// Performs a component-wise operation with another [`Position`].
164    #[inline]
165    fn component_wise_op(&self, other: &Position, op: impl Fn(f32, f32) -> f32) -> Self {
166        Self::new(
167            op(self.x, other.x),
168            op(self.y, other.y),
169            op(self.z, other.z),
170        )
171    }
172
173    /// Performs a scalar operation with a [`Position`] and a [`f32`].
174    #[inline]
175    fn scalar_op(&self, scalar: f32, op: impl Fn(f32, f32) -> f32) -> Self {
176        Self::new(op(self.x, scalar), op(self.y, scalar), op(self.z, scalar))
177    }
178}

301#[cfg(test)]
302mod tests {
303    use super::*;
304    use approx::assert_relative_eq;
305    use std::f32::consts::{FRAC_1_SQRT_2, PI};
306
307    #[test]
308    fn mumble_conversion() {
309        let pos = Position::new(3993.409, 6225.539, -549.570);
310
311        let mumble = pos.to_mumble();
312        assert_relative_eq!(
313            *mumble.as_slice(),
314            [101.433, 13.959, 158.129],
315            max_relative = 1e-3
316        );
317
318        let back = Position::from_mumble(mumble);
319        assert_eq!(back, pos);
320    }
321
322    #[test]
323    fn rotation() {
324        let rotation = Position::new(0.0, 0.25 * PI, 0.5 * PI);
325        let vector = Position::new(1.0, 0.0, 0.0);
326
327        let result = rotation.rotate(vector).to_array();
328        assert_relative_eq!(
329            *result.as_slice(),
330            [0.0, FRAC_1_SQRT_2, -FRAC_1_SQRT_2],
331            max_relative = 1e-7
332        );
333    }
