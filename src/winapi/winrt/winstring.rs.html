extern "system" {
    pub fn WindowsCreateString(

        length: UINT32,
        string: *mut HSTRING,
    ) -> HRESULT;
    pub fn WindowsCreateStringReference(
        sourceString: PCWSTR,

        string: *mut HSTRING,
    ) -> HRESULT;
    pub fn WindowsDeleteString(
        string: HSTRING,
    ) -> HRESULT;
    pub fn WindowsDuplicateString(
        string: HSTRING,
        newString: *mut HSTRING,

    pub fn WindowsGetStringLen(
        string: HSTRING,

        string: HSTRING,
        length: *mut UINT32,
    ) -> PCWSTR;

        string: HSTRING,

    pub fn WindowsStringHasEmbeddedNull(
        string: HSTRING,
        hasEmbedNull: *mut BOOL,
    ) -> HRESULT;
    pub fn WindowsCompareStringOrdinal(
        string1: HSTRING,
        string2: HSTRING,
        result: *mut INT32,
    ) -> HRESULT;
    pub fn WindowsSubstring(
        string: HSTRING,
        startIndex: UINT32,
        newString: *mut HSTRING,
    ) -> HSTRING;
    pub fn WindowsSubstringWithSpecifiedLength(
        string: HSTRING,
        startIndex: UINT32,
        length: UINT32,
        newString: *mut HSTRING,
    ) -> HRESULT;
    pub fn WindowsConcatString(
        string1: HSTRING,
        string2: HSTRING,
        newString: *mut HSTRING,

    ) -> HRESULT;
    pub fn WindowsTrimStringStart(
        string: HSTRING,
        trimString: HSTRING,

    pub fn WindowsTrimStringEnd(
        string: HSTRING,
        trimString: HSTRING,
        newString: *mut HSTRING,
    ) -> HRESULT;
    pub fn WindowsPreallocateStringBuffer(
        length: UINT32,
        charBuffer: *mut *mut WCHAR,
        bufferHandle: *mut HSTRING_BUFFER,
    ) -> HRESULT;
    pub fn WindowsPromoteStringBuffer(
        bufferHandle: HSTRING_BUFFER,
        string: *mut HSTRING,
    ) -> HRESULT;
    pub fn WindowsDeleteStringBuffer(
        bufferHandle: HSTRING_BUFFER,
    ) -> HRESULT;

    pub fn WindowsInspectString(
        targetHString: UINT_PTR,
        machine: USHORT,

        context: *const VOID,
        length: *mut UINT32,
        targetStringAddress: *mut UINT_PTR,

        pFlags: *const ULONG,
        StartingSize: ULONG,
        ppidl: *const HSTRING,
    ) -> ULONG;

        pFlags: *const ULONG,
        pBuffer: *mut UCHAR,
        ppidl: *const HSTRING,

        pFlags: *const ULONG,

        pFlags: *const ULONG,

        pFlags: *const ULONG,

    ) -> *mut UCHAR;
    #[cfg(target_arch = "x86_64")]
    pub fn HSTRING_UserFree64(
