#[inline]
pub fn LISet32(li: &mut LARGE_INTEGER, v: DWORD) {
    unsafe {
        li.u_mut().HighPart = if (v as LONG) < 0 {
            -1
        } else {
            0
        };
        li.u_mut().LowPart = v;
    }
}
#[inline]
pub fn ULISet32(li: &mut ULARGE_INTEGER, v: DWORD) {
    unsafe {
        li.u_mut().HighPart = 0;
        li.u_mut().LowPart = v;
    }
}
